111  bool HitsCompatible(const art::Ptr<recob::Hit>& h1, const art::Ptr<recob::Hit>& h2,
112  const float& _time2cm, const float& _wire2cm, const float& _radius) {
113 
114  if (h1->WireID().Plane != h2->WireID().Plane)
115  return false;
116 
117  double dt = ( h1->PeakTime() - h2->PeakTime() ) * _time2cm;
118  // if the hit time-ranges overlap, this distnce should be 0
119  if (TimeOverlap(h1,h2,_time2cm,dt) == true)
120  dt = 0;
121  double dw = fabs(((double)h1->Channel()-(double)h2->Channel())*_wire2cm);
122  if (dw > 0.3) dw -= 0.3;
123  //if (dw < -0.3) dw = 0.3;
124  double d = dt*dt + dw*dw;
125 
126  if (d > (_radius*_radius))
127  return false;
128 
129  return true;
130  }

252  // a map to connect hit index wih a cluster index
253  // each hit gets a cluster index
254  // _clusterMap[hit_index] -> cluster_index

256  // a map to connect the cluster index with the vector of hit indices for that cluster
257  // _clusters[index] -> vector of hit indices for that cluster
258  std::map<size_t,std::vector<size_t> > _clusters;
259 
260  // keep track of largest cluster ID created
261  size_t maxClusterID = 0;
262 
263  for (int pl=0; pl < 3; pl++){
264 
265  // hit map will only contain hits we want to use for clustering
266  MakeHitMap(hit_ptr_v,pl,_time2cm, _wire2cm, _cellSize, _hitMap);
267 
268  // iterator for hit cell map
269  std::map<std::pair<int,int>, std::vector<size_t> >::iterator it;
270 
271  // loop through hits in each cell to find matches
272  for (it = _hitMap.begin(); it != _hitMap.end(); it++){
273 
274  // pair = (i,j) indices of this cell in the _hitMap
275  auto const& pair = it->first;
276 
277  // wire-space cell index
278  // prepare a hit list of all neighboring cells
279  // _________
280  // |__|__|__|
281  // |__|__|__|
282  // |__|__|__|
283  std::vector<size_t> cellhits = it->second;
284 
285  std::vector<size_t> neighborhits;
286  getNeighboringHits(pair,neighborhits, _hitMap);
287 
288  for (size_t h1=0; h1 < cellhits.size(); h1++){
289 
290  // has this hit been added to a cluster?
291  // if so not necessary to look at
292  auto const& hit1 = cellhits[h1];
293  // keep track if the hit will ever be matched to another
294  bool matched = false;
295  // if not find hits it should be clustered with and add it to the appropriate cluster
296  for (size_t h2=0; h2 < neighborhits.size(); h2++){
297  auto const& hit2 = neighborhits[h2];
298  if (hit1 == hit2) continue;
299  // are the hits compatible?
300  bool compat = HitsCompatible(hit_ptr_v.at(hit1),
301  hit_ptr_v.at(hit2),
302  _time2cm, _wire2cm, _radius);
303  // should the hits go in the same cluster?
304  if (compat){
305  matched = true;
306  // if both hits have already been assigned to a cluster then we can merge the cluster indices!
307  if ( (_clusterMap.find(hit1) != _clusterMap.end()) and
308  (_clusterMap.find(hit2) != _clusterMap.end()) ){
309  // if in the same cluster -> do nothing
310  // if they are in different clusters:
311  if (_clusterMap[hit1] != _clusterMap[hit2]){
312  auto idx1 = _clusterMap[hit1];
313  auto idx2 = _clusterMap[hit2];
314  // hit indices for 1st cluster:

316  auto hits2 = _clusters[idx2];

319  hits1.push_back(h);
320  // also change the index that the hit goes to (idx1 instead of idx2)
321  _clusterMap[h] = idx1;
322  }
323  _clusters[idx1] = hits1;
324  // erase cluster @ index2
325  _clusters.erase(idx2);
326  }// if they are in different clusters
327  }
328  // if compatible and the 2nd hit has been added to a cluster
329  // add hit1 to the same cluster
330  else if ( (_clusterMap.find(hit2) != _clusterMap.end()) and
331  (_clusterMap.find(hit1) == _clusterMap.end()) ){
332  auto clusIdx = _clusterMap[hit2];
333  _clusterMap[hit1] = clusIdx;
334  _clusters[clusIdx].push_back(hit1);
335  }
336  // otherwise, add both to a new cluster
337  else if ( (_clusterMap.find(hit1) != _clusterMap.end()) and
338  (_clusterMap.find(hit2) == _clusterMap.end()) ){
339  auto clusIdx = _clusterMap[hit1];
340  _clusterMap[hit2] = clusIdx;
341  _clusters[clusIdx].push_back(hit2);
342  }
343  // if neither has a cluster yet
344  else{
345  // create a new cluster for this match
346  _clusterMap[hit1] = maxClusterID;
347  _clusterMap[hit2] = maxClusterID;
348  std::vector<size_t> cl = {hit1,hit2};
349  _clusters[maxClusterID] = cl;
350  maxClusterID += 1;
351  }
352  }// if the two hits are compatible
353  }// 2nd loop through hits in the cell
354  // has this hit been matched? if not we still need to add it as its own cluster
355  if (matched == false){
356  _clusterMap[hit1] = maxClusterID;
357  _clusters[maxClusterID] = {hit1};
358  maxClusterID += 1;
359  }
360  }// 1st loop through hits in the cell
361  }// loop through all cells
362 
363  }// loop through all planes
364 
365  // make a vector for the clusters
366  for (auto it = _clusters.begin(); it != _clusters.end(); it++){
367  auto indices = it->second;
368  // if there are enough indices, make a cluster
369  if (indices.size() >= 1){
370  std::vector<unsigned int> clus;
371  for (auto idx : indices)
372  clus.push_back(idx);
373  _out_cluster_vector.push_back(clus);
374  }// if there are 2 hits in cluster
375  }
376 
377  return true;
378  }
